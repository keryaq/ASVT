$MOD51

ADC_PORT    EQU P1
DAC_L       EQU P0
DAC_H       EQU P2
ADC_SOC     BIT P3.6
ADC_nEOC    BIT P3.7
MODE_FLAG   BIT 00H

; ???????? +8V: 93 = 0x5D
OFFSET_8    EQU 05FH

ORG 0000H
LJMP MAIN
ORG 0003H
LJMP INT0_ISR

ORG 0030H
MAIN:
    MOV SP, #60H
    CLR MODE_FLAG
    CLR ADC_SOC
    SETB EA
    SETB EX0
    SETB IT0
    MOV DAC_L, #00H
    MOV DAC_H, #00H

LOOP:
    SETB ADC_SOC

WAIT_ADC:
    JB ADC_nEOC, WAIT_ADC

    MOV A, ADC_PORT
    CLR ADC_SOC
    
    JNB MODE_FLAG, SCALE
    
    ADD A, #OFFSET_8
    JNC SCALE
    MOV A, #0FFH

SCALE:
    MOV DAC_H, A
    MOV DAC_L, #00H

    SJMP LOOP

INT0_ISR:
    CPL MODE_FLAG
    RETI

END
